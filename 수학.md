``` c   
  int gcd(int a,int b){
    return b?gcd(b,a%b):a;
  }
   
```
# 유클리드 확장
``` c   
  int nA;
  int MOD_INV(int mod, int y){
      if(y==0){
          nA=mod;
          return 0;
      }
      int r=mod/y,A=MOD_INV(y,mod%y);
      int B=nA-(A*r);
      nA=A;
      return B;
  }
   
```
